Integrate Daemo API

from daemo.client import DaemoClient

def rate_workers(worker_responses):
    ratings = [{
                   "task_id": worker_response.get("task_id"),
                   "worker_id": worker_response.get("worker_id"),
                   "weight": 3 if is_correct(response) else 1
               } for response in worker_responses]

    # rate workers to improve quality with every iteration
    client.rate(project_key=PROJECT_KEY, ratings=ratings)

client = DaemoClient()

client.publish(
    project_key=PROJECT_KEY,
    tasks=[{"tweet": "Some tweet text."}],
    approve=approve_correct_response,
    completed=rate_workers
)
